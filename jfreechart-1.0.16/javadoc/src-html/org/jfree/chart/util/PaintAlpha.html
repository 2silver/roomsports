058    /**
059     * This class contains static methods for the manipulation
060     * of objects of type <code>Paint</code>
061     * <p>
062     * The intention is to honour the alpha-channel in the process.
063     * <code>PaintAlpha</code> was originally conceived to improve the
064     * rendering of 3D Shapes with transparent colours and to allow
065     * invisible bars by making them completely transparent.
066     * <p>
067     * In, for example
068     * {@link org.jfree.chart.renderer.category.StackedBarRenderer3D StackedBarRenderer3D},
069     * bars are rendered with 6 faces. The front face is rendered with
070     * the <code>Paint</code> requested. The other 5 faces are rendered
071     * darker to achieve the 3D effect.
072     * <p>
073     * Previously {@link Color#darker()} was used for this,
074     * which always returns an opaque colour.
075     * <p>
076     * Additionally there are methods to control the behaviour and
077     * in particular a {@link PaintAlpha#cloneImage(BufferedImage) cloneImage(..)}
078     * method which is needed to darken objects of type {@link TexturePaint}.
079     *
080     * @author  DaveLaw
081     */
082    public class PaintAlpha {

089         * Multiplier for the <code>darker</code> Methods.<br>
090         * (taken from {@link java.awt.Color}.FACTOR)

096        /**
097         * Per default <code>PaintAlpha</code> will try to honour alpha-channel
098         * information.  In the past this was not the case.
099         * If you wish legacy functionality for your application you can request
100         * this here.
101         *
102         * @param legacyAlpha boolean
103         *
104         * @return the previous setting
105         */
106        public static boolean setLegacyAlpha(boolean legacyAlpha) {
107            boolean old = PaintAlpha.legacyAlpha;
108            PaintAlpha.legacyAlpha = legacyAlpha;
109            return old;
110        }

112        /**
113         * Create a new (if possible, darker) <code>Paint</code> of the same Type.
114         * If the Type is not supported, the original <code>Paint</code> is returned.
115         * <p>
116         * @param paint a <code>Paint</code> implementation
117         * (e.g. {@link Color}, {@link GradientPaint}, {@link TexturePaint},..)
118         * <p>
119         * @return a (usually new, see above) <code>Paint</code>
120         */
121        public static Paint darker(Paint paint) {
122    
123            if (paint instanceof Color) {
124                return darker((Color) paint);
125            }
126            if (legacyAlpha == true) {
127                /*
128                 * Legacy? Just return the original Paint.
129                 * (this corresponds EXACTLY to how Paints used to be darkened)
130                 */
131                return paint;
132            }
133            if (paint instanceof GradientPaint) {
134                return darker((GradientPaint) paint);
135            }
136            if (paint.getClass().getName().equals("java.awt.LinearGradientPaint")) {
137                // TODO -> instanceof
138                return darkerLinearGradientPaint(paint);
139            }
140            if (paint.getClass().getName().equals("java.awt.RadialGradientPaint")) {
141                // TODO -> instanceof
142                return darkerRadialGradientPaint(paint);
143            }
144            if (paint instanceof TexturePaint) {
145                return darker((TexturePaint) paint, true);
146            }
147            return paint;
148        }

150        /**
151         * Similar to {@link Color#darker()}.
152         * <p>
153         * The essential difference is that this method
154         * maintains the alpha-channel unchanged<br>
155         *
156         * @param paint a <code>Color</code>
157         *
158         * @return a darker version of the <code>Color</code>
159         */
160        private static Color darker(Color paint) {
161            return new Color(
162                    (int)(paint.getRed  () * FACTOR),
163                    (int)(paint.getGreen() * FACTOR),
164                    (int)(paint.getBlue () * FACTOR), paint.getAlpha());
165        }

280        /**
281         * Convenience method to invoke the zero argument <code>methodName</code>
282         * method of <code>object</code> via Reflection.
283         *
284         * @param object
285         * @param methodName
286         *
287         * @return the result
288         *
289         * @throws IllegalArgumentException
290         * @throws SecurityException
291         * @throws IllegalAccessException
292         * @throws InvocationTargetException
293         * @throws NoSuchMethodException
294         */
295        private static final Object invokeZeroArgumentMethod(Object object,
296                String methodName) throws IllegalArgumentException,
297                SecurityException, IllegalAccessException,
298                InvocationTargetException, NoSuchMethodException {
299            return object.getClass().getMethod(methodName,
300                    new Class[] {}).invoke(object, new Object[] {});
301        }

303        /**
304         * Create a new <code>TexturePaint</code> with its colours darkened.
305         * <p>
306         * This entails cloning the underlying <code>BufferedImage</code>,
307         * then darkening each colour-pixel individually!
308         *
309         * @param paint a <code>TexturePaint</code>
310         *
311         * @param ignoreThisDummyArgument which is just to guarantee a unique
312         *        method signature
313         *
314         * @return a darker version of the <code>TexturePaint</code>
315         */
316        private static TexturePaint darker(TexturePaint paint,
317                boolean ignoreThisDummyArgument) {
318            try {
319                return darker(paint);
320            }
321            catch (Exception e) {
322                /*
323                 * Lots can go wrong while fiddling with Images, Colour Models
324                 * & such!  If anything at all goes awry, just return the original
325                 * TexturePaint.  (TexturePaint's are immutable anyway, so no harm
326                 * done)
327                 */
328                return paint;
329            }
330        }

332        /**
333         * Create a new <code>TexturePaint</code> with its colours darkened.
334         * <p>
335         * This entails cloning the underlying <code>BufferedImage</code>,
336         * then darkening each colour-pixel individually!
337         *
338         * @param paint a <code>TexturePaint</code>
339         *
340         * @return a darker version of the <code>TexturePaint</code>
341         */
342        private static TexturePaint darker(TexturePaint paint) {
343            /**
344             * Colour Models with pre-multiplied Alpha tested OK without any
345             * special logic
346             *
347             * BufferedImage.TYPE_INT_ARGB_PRE:    // Type 03: tested OK 2011.02.27
348             * BufferedImage.TYPE_4BYTE_ABGR_PRE:  // Type 07: tested OK 2011.02.27
349             */
350            if (paint.getImage().getColorModel().isAlphaPremultiplied()) {
351                /* Placeholder */
352            }
353    
354            BufferedImage img = cloneImage(paint.getImage());
355    
356            WritableRaster ras = img.copyData(null);
357    
358            final int miX = ras.getMinX();
359            final int miY = ras.getMinY();
360            final int maY = ras.getMinY() + ras.getHeight();
361    
362            final int   wid = ras.getWidth();
363    
364            /**/  int[] pix = new int[wid * img.getSampleModel().getNumBands()];
365            /* (pix-buffer is large enough for all pixels of one row) */
366    
367            /**
368             * Indexed Colour Models (sort of a Palette) CANNOT be simply
369             * multiplied (the pixel-value is just an index into the Palette).
370             *
371             * Fortunately, IndexColorModel.getComponents(..) resolves the colours.
372             * The resolved colours can then be multiplied by our FACTOR.
373             * IndexColorModel.getDataElement(..) then tries to map the computed
374             * colour to the "nearest" in the Palette.
375             *
376             * It is quite possible that the "nearest" colour is the ORIGINAL
377             * colour!  In the worst case, the returned Image will be identical to
378             * the original.
379             *
380             * Applies to following Image Types:
381             *
382             * BufferedImage.TYPE_BYTE_BINARY:    // Type 12: tested OK 2011.02.27
383             * BufferedImage.TYPE_BYTE_INDEXED:   // Type 13: tested OK 2011.02.27
384             */
385            if (img.getColorModel() instanceof IndexColorModel) {
386    
387                int[] nco = new int[4]; // RGB (+ optional Alpha which we leave
388                                        // unchanged)
389    
390                for (int y = miY; y < maY; y++)  {
391    
392                    pix = ras.getPixels(miX, y, wid, 1, pix);
393    
394                    for (int p = 0; p < pix.length; p++) {
395                        nco    =  img.getColorModel().getComponents(pix[p], nco, 0);
396                        nco[0] *= FACTOR; // Red
397                        nco[1] *= FACTOR; // Green
398                        nco[2] *= FACTOR; // Blue. Now map computed colour to
399                                          // nearest in Palette...
400                        pix[p] = img.getColorModel().getDataElement(nco, 0);
401                    }
402                    /**/ ras.setPixels(miX, y, wid, 1, pix);
403                }
404                img.setData(ras);
405    
406                return new TexturePaint(img, paint.getAnchorRect());
407            }
408    
409            /**
410             * For the other 2 Colour Models, java.awt.image.ComponentColorModel and
411             * java.awt.image.DirectColorModel, the order of subpixels returned by
412             * ras.getPixels(..) was observed to correspond to the following...
413             */
414            if (img.getSampleModel().getNumBands() == 4) {
415                /**
416                 * The following Image Types have an Alpha-channel which we will
417                 * leave unchanged:
418                 *
419                 * BufferedImage.TYPE_INT_ARGB:   // Type 02: tested OK 2011.02.27
420                 * BufferedImage.TYPE_4BYTE_ABGR: // Type 06: tested OK 2011.02.27
421                 */
422                for (int y = miY; y < maY; y++)  {
423    
424                    pix = ras.getPixels(miX, y, wid, 1, pix);
425    
426                    for (int p = 0; p < pix.length;) {
427                        pix[p] = (int)(pix[p++] * FACTOR); // Red
428                        pix[p] = (int)(pix[p++] * FACTOR); // Green
429                        pix[p] = (int)(pix[p++] * FACTOR); // Blue
430                        /* Ignore alpha-channel -> */p++;
431                    }
432                    /**/  ras.setPixels(miX, y, wid, 1, pix);
433                }
434                img.setData(ras);
435                return new TexturePaint(img, paint.getAnchorRect());
436            } else {
437                for (int y = miY; y < maY; y++)  {
438    
439                    pix = ras.getPixels(miX, y, wid, 1, pix);
440    
441                    for (int p = 0; p < pix.length; p++) {
442                        pix[p] = (int)(pix[p] * FACTOR);
443                    }
444                    /**/  ras.setPixels(miX, y, wid, 1, pix);
445                }
446                img.setData(ras);
447                return new TexturePaint(img, paint.getAnchorRect());
448                /**
449                 * Above, we multiplied every pixel by our FACTOR because the
450                 * applicable Image Types consist only of colour or grey channels:
451                 *
452                 * BufferedImage.TYPE_INT_RGB:        // Type 01: tested OK 2011.02.27
453                 * BufferedImage.TYPE_INT_BGR:        // Type 04: tested OK 2011.02.27
454                 * BufferedImage.TYPE_3BYTE_BGR:      // Type 05: tested OK 2011.02.27
455                 * BufferedImage.TYPE_BYTE_GRAY:      // Type 10: tested OK 2011.02.27
456                 * BufferedImage.TYPE_USHORT_GRAY:    // Type 11: tested OK 2011.02.27
457                 * BufferedImage.TYPE_USHORT_565_RGB: // Type 08: tested OK 2011.02.27
458                 * BufferedImage.TYPE_USHORT_555_RGB: // Type 09: tested OK 2011.02.27
459                 *
460                 * Note: as ras.getPixels(..) returned colours in the order R, G, B, A (optional)
461                 * for both TYPE_4BYTE_ABGR & TYPE_3BYTE_BGR,
462                 * it is assumed that TYPE_INT_BGR will behave similarly.
463                 */
464            }
465        }

467        /**
468         * Clone a {@link BufferedImage}.
469         * <p>
470         * Note: when constructing the clone, the original Colour Model Object is
471         * reused.<br>  That keeps things simple & should not be a problem, as all
472         * known Colour Models<br>
473         * ({@link java.awt.image.IndexColorModel     IndexColorModel},
474         *  {@link java.awt.image.DirectColorModel    DirectColorModel},
475         *  {@link java.awt.image.ComponentColorModel ComponentColorModel}) are
476         * immutable.
477         *
478         * @param image original BufferedImage to clone
479         *
480         * @return a new BufferedImage reusing the original's Colour Model &
481         *         containing a clone of its pixels
482         */
483        public static BufferedImage cloneImage(BufferedImage image) {
484    
485            WritableRaster rin = image.getRaster();
486            WritableRaster ras = rin.createCompatibleWritableRaster();
487            /**/ ras.setRect(rin); // <- this is the code that actually COPIES the pixels
488    
489            /*
490             * Buffered Images may have properties, but NEVER disclose them!
491             * Nevertheless, just in case someone implements getPropertyNames()
492             * one day...
493             */
494            Hashtable props = null;
495            String[] propNames = image.getPropertyNames();
496            if (propNames != null) { // ALWAYS null
497                props = new Hashtable();
498                for (int i = 0; i < propNames.length; i++) {
499                    props.put(propNames[i], image.getProperty(propNames[i]));
500                }
501            }
502            return new BufferedImage(image.getColorModel(), ras,
503                    image.isAlphaPremultiplied(), props);
504        }
